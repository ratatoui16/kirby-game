Kirby-Style Platformer Game

This is a complete single-file React game inspired by classic 2D Kirby platformers. Features include:

  • 2D side-scrolling platformer gameplay
  • Jumping and gravity physics
  • Enemy swallowing mechanic
  • Power stealing system
  • Fire and sword abilities
  • Projectile attacks
  • Health system
  • Animated enemies
{String.raw`import React, { useEffect, useRef, useState } from "react";

const WIDTH = 1000;
const HEIGHT = 560;
const GROUND = 480;

const PLAYER_SPEED = 4;
const JUMP_FORCE = -12;
const GRAVITY = 0.55;

const enemyTemplates = [
  {
    id: 1,
    x: 420,
    y: 440,
    type: "fire",
    color: "orange",
    dir: -1,
  },
  {
    id: 2,
    x: 700,
    y: 440,
    type: "sword",
    color: "deepskyblue",
    dir: 1,
  },
  {
    id: 3,
    x: 980,
    y: 440,
    type: "none",
    color: "limegreen",
    dir: -1,
  },
];

export default function Game() {
  const canvasRef = useRef(null);

  const [player, setPlayer] = useState({
    x: 120,
    y: 380,
    w: 42,
    h: 42,
    vx: 0,
    vy: 0,
    onGround: false,
    facing: 1,
    power: "none",
    hp: 5,
    inhaling: false,
  });

  const [enemies, setEnemies] = useState(enemyTemplates);
  const [projectiles, setProjectiles] = useState([]);
  const [keys, setKeys] = useState({});

  useEffect(() => {
    const down = (e) => {
      setKeys((k) => ({ ...k, [e.key.toLowerCase()]: true }));
    };

    const up = (e) => {
      setKeys((k) => ({ ...k, [e.key.toLowerCase()]: false }));
    };

    window.addEventListener("keydown", down);
    window.addEventListener("keyup", up);

    return () => {
      window.removeEventListener("keydown", down);
      window.removeEventListener("keyup", up);
    };
  }, []);

  useEffect(() => {
    const interval = setInterval(gameLoop, 1000 / 60);
    return () => clearInterval(interval);
  });

  function shoot(power, x, y, dir) {
    if (power === "none") return;

    setProjectiles((p) => [
      ...p,
      {
        id: Math.random(),
        x,
        y,
        dir,
        power,
      },
    ]);
  }

  function gameLoop() {
    setPlayer((p) => {
      let vx = 0;
      let vy = p.vy + GRAVITY;
      let x = p.x;
      let y = p.y;
      let facing = p.facing;
      let inhaling = false;

      if (keys["a"] || keys["arrowleft"]) {
        vx = -PLAYER_SPEED;
        facing = -1;
      }

      if (keys["d"] || keys["arrowright"]) {
        vx = PLAYER_SPEED;
        facing = 1;
      }

      if ((keys["w"] || keys[" "] || keys["arrowup"]) && p.onGround) {
        vy = JUMP_FORCE;
      }

      if (keys["j"]) {
        inhaling = true;

        setEnemies((enemyList) => {
          const updated = enemyList.filter((enemy) => {
            const dx = enemy.x - p.x;
            const dy = Math.abs(enemy.y - p.y);

            const close = Math.abs(dx) < 70 && dy < 50;
            const inFront = facing === 1 ? dx > 0 : dx < 0;

            if (close && inFront) {
              setPlayer((old) => ({
                ...old,
                power: enemy.type,
              }));
              return false;
            }

            return true;
          });

          return updated;
        });
      }

      if (keys["k"]) {
        shoot(p.power, p.x + 20, p.y + 20, facing);
      }

      x += vx;
      y += vy;

      let onGround = false;

      if (y + p.h >= GROUND) {
        y = GROUND - p.h;
        vy = 0;
        onGround = true;
      }

      return {
        ...p,
        x,
        y,
        vx,
        vy,
        facing,
        onGround,
        inhaling,
      };
    });

    setEnemies((enemyList) =>
      enemyList.map((e) => {
        let x = e.x + e.dir * 1.1;
        let dir = e.dir;

        if (x < 300) dir = 1;
        if (x > 1400) dir = -1;

        return {
          ...e,
          x,
          dir,
        };
      })
    );

    setProjectiles((shots) =>
      shots
        .map((s) => ({
          ...s,
          x: s.x + s.dir * 7,
        }))
        .filter((s) => s.x > -100 && s.x < 3000)
    );

    setEnemies((enemyList) => {
      return enemyList.filter((enemy) => {
        let hit = false;

        projectiles.forEach((shot) => {
          const overlap =
            shot.x < enemy.x + 34 &&
            shot.x + 12 > enemy.x &&
            shot.y < enemy.y + 34 &&
            shot.y + 12 > enemy.y;

          if (overlap) {
            hit = true;
          }
        });

        return !hit;
      });
    });

    draw();
  }

  function draw() {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext("2d");

    ctx.clearRect(0, 0, WIDTH, HEIGHT);

    ctx.fillStyle = "#9be7ff";
    ctx.fillRect(0, 0, WIDTH, HEIGHT);

    ctx.fillStyle = "#6bd16b";
    ctx.fillRect(0, GROUND, WIDTH, HEIGHT - GROUND);

    ctx.fillStyle = "#4a9f4a";

    for (let i = 0; i < WIDTH; i += 64) {
      ctx.fillRect(i, GROUND - 10, 64, 10);
    }

    ctx.fillStyle = "pink";
    ctx.beginPath();
    ctx.arc(player.x + 21, player.y + 21, 21, 0, Math.PI * 2);
    ctx.fill();

    ctx.fillStyle = "black";
    ctx.beginPath();
    ctx.arc(player.x + 16, player.y + 17, 2, 0, Math.PI * 2);
    ctx.fill();

    ctx.beginPath();
    ctx.arc(player.x + 26, player.y + 17, 2, 0, Math.PI * 2);
    ctx.fill();

    if (player.inhaling) {
      ctx.strokeStyle = "white";
      ctx.lineWidth = 4;

      const mouthX = player.facing === 1 ? player.x + 40 : player.x + 2;

      ctx.beginPath();
      ctx.moveTo(mouthX, player.y + 21);
      ctx.lineTo(mouthX + player.facing * 50, player.y + 21);
      ctx.stroke();
    }

    if (player.power === "fire") {
      ctx.fillStyle = "orange";
      ctx.fillRect(player.x + 8, player.y - 8, 26, 8);
    }

    if (player.power === "sword") {
      ctx.fillStyle = "silver";
      ctx.fillRect(player.x + 32, player.y + 12, 22, 5);
    }

    enemies.forEach((enemy) => {
      ctx.fillStyle = enemy.color;
      ctx.beginPath();
      ctx.arc(enemy.x + 17, enemy.y + 17, 17, 0, Math.PI * 2);
      ctx.fill();

      ctx.fillStyle = "black";
      ctx.fillRect(enemy.x + 10, enemy.y + 12, 3, 3);
      ctx.fillRect(enemy.x + 20, enemy.y + 12, 3, 3);
    });

    projectiles.forEach((shot) => {
      ctx.fillStyle = shot.power === "fire" ? "red" : "white";

      ctx.beginPath();
      ctx.arc(shot.x, shot.y, 8, 0, Math.PI * 2);
      ctx.fill();
    });

    ctx.fillStyle = "black";
    ctx.font = "20px Arial";
    ctx.fillText("Power: " + player.power, 20, 30);
    ctx.fillText("Controls: A/D Move | W Jump | J Inhale | K Attack", 20, 55);
  }

  return (
    

      
    

  );
}
`}

{String.raw`npm install gh-pages --save-dev`}

{String.raw`{

    "build": "vite build",
    "deploy": "gh-pages -d dist"

}`}

{String.raw`import { defineConfig } from 'vite'

})`}
